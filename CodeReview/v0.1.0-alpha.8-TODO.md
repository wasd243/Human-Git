async fn get_repo_path(
    repo_path: Option<String>,
    state: &AppState,
) -> String {
    match repo_path {
        Some(p) if !p.is_empty() && p != "." => p,
        _ => {
            let current = state.current_repo_path.lock().await;
            current
                .as_ref()
                .map(|p| p.to_string_lossy().to_string())
                .unwrap_or_else(|| ".".to_string())
        }
    }
}
